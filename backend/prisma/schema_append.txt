model VegetableRequest {
  id              Int              @id @default(autoincrement())
  vegetable_name  String
  requested_by    Int
  status          VegetableStatus  @default(PENDING)
  admin_notes     String?
  created_at      DateTime         @default(now())
  updated_at      DateTime         @updatedAt

  // Relations
  farmer User @relation("VegetableRequests", fields: [requested_by], references: [id], onDelete: Cascade)

  @@map("vegetable_requests")
}
